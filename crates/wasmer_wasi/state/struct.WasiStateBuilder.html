Convenient builder API for configuring WASI via WasiState.

Usage:

let mut state_builder = WasiState::new("wasi-prog-name");
state_builder
   .env("ENV_VAR", "ENV_VAL")
   .arg("--verbose")
   .preopen_dir("src")?
   .map_dir("name_wasi_sees", "path/on/host/fs")?
   .build();
